function getMonthlyBiorhythm(kuboType) { const monthlyData = []; const startYear = 2025; const endYear = 2036; // 空亡タイプごとの月の設定（安全な処理） const kuboMonths = { '子丑': { odd: [1, 3, 5, 7, 9, 11], even: [2, 4, 6, 8, 10, 12] }, '寅卯': { odd: [2, 4, 6, 8, 10, 12], even: [1, 3, 5, 7, 9, 11] }, '辰巳': { odd: [1, 3, 5, 7, 9, 11], even: [2, 4, 6, 8, 10, 12] }, '午未': { odd: [2, 4, 6, 8, 10, 12], even: [1, 3, 5, 7, 9, 11] }, '申酉': { odd: [1, 3, 5, 7, 9, 11], even: [2, 4, 6, 8, 10, 12] }, '戌亥': { odd: [2, 4, 6, 8, 10, 12], even: [1, 3, 5, 7, 9, 11] } }; // 空亡タイプのデータが存在しない場合のデフォルト値 const monthSettings = kuboMonths[kuboType] || { odd: [], even: [] }; for (let year = startYear; year <= endYear; year++) { const yearData = { year: year, months: [] }; const isOddYear = year % 2 === 1; const goodMonths = isOddYear ? monthSettings.odd : monthSettings.even; for (let month = 1; month <= 12; month++) { yearData.months.push({ month: month, isGood: goodMonths.includes(month) }); } monthlyData.push(yearData); } return monthlyData; }
